\documentclass{article}
\usepackage{tikz}

\newcommand{\sectionbox}[1]{%
  \tikz[overlay,remember picture,baseline={(current bounding box.south)}]
  \node[fill=gray!20,inner sep=0pt,outer sep=0pt,minimum width=\textwidth,yshift=-1em]
    at (current page.west) {\phantom{#1}};}

\newcommand{\subsectionbox}[1]{%
  \tikz[overlay,remember picture,baseline={(current bounding box.south)}]
  \node[fill=red!20,inner sep=0pt,outer sep=0pt,minimum width=\textwidth,yshift=-1em]
    at (current page.west) {\phantom{#1}};}
